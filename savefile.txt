SaveGame(FileInput file) {
    file.writeInt(playerCoin);
    file.writeInt(playerEquipment);
    file.writeInt(gameDay);
}

LoadGame(FileInput file) {
    if(file.exists()) {
        playerCoin= file.readInt();
        playerEquipment = file.readInt();
        gameDay = file.readInt();
    } else {
        playerCoin = 0;
        playerEquipment = 0;
        gameDay = 0;
    }
}
